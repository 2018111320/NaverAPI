*도커 설치
*net-tools 설치

[Dockerfile 작성]
sudo vim Dockerfile

FROM nginx:latest

WORKDIR /etc/nginx

COPY default.conf /etc/nginx/conf.d/default.conf

CMD ["nginx", "-g", "daemon off;"]

EXPOSE 80
EXPOSE 443

[default.conf 작성]
sudo vim default.conf

server {
        listen 80;
        server_name ~.;

        location / {
                proxy_pass http://:5001;
        }
}

도커 이미지 빌드
sudo docker build -t linux:nginx .

sudo docker images

도커 실행
sudo docker run -d --name linux_nginx -p 80:80 linux:nginx

sudo docker ps

netstat -tnlp

[도커 자동 실행 서비스 작성]
sudo vi /etc/systemd/system/linux_nginx.service

[Unit]
Wants=docker.service
After=docker.service

[Service]
RemainAfterExit=yes
ExecStart=/usr/bin/docker start linux_nginx
ExecStop=/usr/bin/docker stop linux_nginx

[Install]
WantedBy=multi-user.target

도커 자동 실행 서비스 활성화
sudo systemctl enable linux_nginx.service
